[docs]        def exposed_execute(self):
            """Start inputsream.

            This function is supposed to be wrapped by `rpyc.async`
            """
            stream = InputStreamExecutorService._create_stream(self._istream_name, self._istream_args)
            for batch in stream:
                InputStreamExecutorService._store_new_batch(batch)
                self._on_new_batch(self._conn, batch)


    @staticmethod
    def _create_stream(istream_name, istream_args):
        """Create stream object"""
        module       = import_module(InputStreamExecutorService._get_module_name(istream_name))
        stream_class = getattr(module, istream_name)
        stream       = stream_class(*istream_args)
        return stream

        """Return full module path of `istream_name`"""
        module = 'shellstreaming.istream.%s' % (istream_name.lower())

[docs]    def join(self):
        """Wait for istream to finish.

        After istream finishes, connection to worker is closed.

        :raises: any exception raised from `_async_execute`
        """
        self._async_res.wait()
        if self._async_res.error:  # pragma: no cover
            raise self._async_res.value

        (worker, connection, conn_thread) = self._conn
        conn_thread.stop()
        connection.close()



def _connect(worker):
    """Connect to worker"""
    connection  = rpyc.connect(worker, int(Config.instance().get('worker', 'port')))
    conn_thread = rpyc.BgServingThread(connection)
    return (worker, connection, conn_thread)


def _async_execute(conn, istream_name, istream_args):
    """Asynchronously execute istream on worker process"""
    (worker, connection, conn_thread) = conn
    executor = connection.root.InputStreamExecutor(
        conn,
        istream_name, istream_args,
        on_new_batch=_reg_new_batch
    )

    # asynchronously call `InputStreamExecutor.exposed_execute`, in which callbacks are called
    aexecute = rpyc.async(executor.execute)
    return aexecute()
