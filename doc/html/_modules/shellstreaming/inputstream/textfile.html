[docs]    def run(self):
        """Reads a text file line-by-line until EOF"""
        with open(self._path) as f:
            rdef = RecordDef([{'name': 'line', 'type': 'STRING'}])
            line = f.readline()
            while line:
                self.add(Record(rdef, line))
                line = f.readline()
        self.add(None)  # producer has end data-fetching
